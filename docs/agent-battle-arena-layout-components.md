1.2 布局组件说明

区域 尺寸 功能描述

Header 80px 显示系统名称、当前议题、计时器

Agent Cluster 自适应 弧形排列的Agent节点，显示角色状态

interface AgentNodeProps {

 id: string;

 name: string; // 如 \"架构师Agent\"

 character: PixelCharacter; // 像素角色枚举

 color: AgentColor; // 专属颜色

 hp: number; // 0-100 (像素块血量)

 energy: number; // 0-100 (能量豆进度)

 score: number; // 累计得分

 status: \'idle\' \| \'thinking\' \| \'attacking\' \| \'damaged\' \| \'powered\' \| \'defeated\';

 isCurrentTurn: boolean; // 是否当前回合
